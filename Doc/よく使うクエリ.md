- ExternalIntegration取込用
```
use OpenPlatform
update Orders
set 
    IsExportable = 1
    , UpdateTime = GETDATE()    
Where
    OrderId IN(SELECT OrderId FROM OrderDeliveries WHERE PBOrderNo = @OrderNo)
```

- Orders注文状況
```
SELECT * FROM Orders WHERE OrderId IN (SELECT OrderId FROM OrderDeliveries WHERE PBOrderNo = @OrderNo)
SELECT * FROM OrderDeliveries WHERE PBOrderNo = @OrderNo
```

- Photoback注文状況
```
use photoback
select * from TBLOrders O
inner join TBLOrderPhotobacks OP on O.OrderNo = OP.OrderNo
inner join TBLPhotobacks P on OP.ContentID = P.ContentID
Where O.OrderNo = @OrderNo

update TBLPhotobacks
set PdfStatus = 1,UpdateTime = GETDATE()
Where ContentID = 'CWIP-5016310000000141783'
```
